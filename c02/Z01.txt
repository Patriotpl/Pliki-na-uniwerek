int absolute(int x)
{
    if (x<0)
    {
        return -x;
    }
    else
    {
        return x;
    }
}

int silnia(int x)
{
    int wynik = 1;
    int i;

    for (i = 0; i<x; i++)
        {
            wynik*=i+1;
        }
    return wynik;
}

int nwd(int x)
{
    int i;
    for (i = x - 1; i>0; i--)
    {
        if(x%i == 0)
        {
            return i;
        }
    }
    return 1;
}

int pot2(unsigned int x)

float pot22(int x)
{
    float wynik = 1;
    int i;
    if (x>=0)
    {
        for (i = 0; i<x; i++)

            wynik*=2;

{

}
